 24: namespace Thelia\Core\Template\Assets;
 25: 
 26: use Assetic\AssetManager;
 27: use Assetic\FilterManager;
 28: use Assetic\Filter;
 29: use Assetic\Factory\AssetFactory;
 30: use Assetic\Factory\Worker\CacheBustingWorker;
 31: use Assetic\AssetWriter;
 32: use Assetic\Asset\AssetCache;
 33: use Assetic\Cache\FilesystemCache;
 34: 
 35: /**
 36:  * This class is a simple helper for generating assets using Assetic.
 37:  *
 38:  * @author Franck Allimant <franck@cqfdev.fr>
 39:  */
 40: class AsseticHelper
 41: {
 42:     /**
 43:      * Generates assets from $asset_path in $output_path, using $filters.
 44:      *
 45:      * @param  string                    $asset_path  the full path to the asset file (or file collection)
 46:      * @param  unknown                   $output_path the full disk path to the output directory (shoud be visible to web server)
 47:      * @param  unknown                   $output_url  the URL to the generated asset directory
 48:      * @param  unknown                   $asset_type  the asset type: css, js, ... The generated files will have this extension. Pass an empty string to use the asset source extension.
 49:      * @param  unknown                   $filters     a list of filters, as defined below (see switch($filter_name) ...)
 50:      * @param  unknown                   $debug       true / false
 51:      * @throws \InvalidArgumentException if an invalid filter name is found
 52:      * @return string                    The URL to the generated asset file.
 53:      */
 54:     public function asseticize($asset_path, $output_path, $output_url, $asset_type, $filters, $debug)
 55:     {
 56:         $asset_name = basename($asset_path);
 57:         $asset_dir = dirname($asset_path);
 58: 
 59:         $am = new AssetManager();
 60:         $fm = new FilterManager();
 61: 
 62:         if (! empty($filters)) {
 63:             $filter_list = explode(',', $filters);
 64: 
 65:             foreach ($filter_list as $filter_name) {
 66: 
 67:                 $filter_name = trim($filter_name);
 68: 
 69:                 switch ($filter_name) {
 70:                     case 'less' :
 71:                         $fm->set('less', new Filter\LessphpFilter());
 72:                         break;
 73: 
 74:                     case 'sass' :
 75:                         $fm->set('less', new Filter\Sass\SassFilter());
 76:                         break;
 77: 
 78:                     case 'cssembed' :
 79:                         $fm->set('cssembed', new Filter\PhpCssEmbedFilter());
 80:                         break;
 81: 
 82:                     case 'cssrewrite':
 83:                         $fm->set('cssrewrite', new Filter\CssRewriteFilter());
 84:                         break;
 85: 
 86:                     case 'cssimport':
 87:                         $fm->set('cssimport', new Filter\CssImportFilter());
 88:                         break;
 89: 
 90:                     default :
 91:                         throw new \InvalidArgumentException("Unsupported Assetic filter: '$filter_name'");
 92:                         break;
 93:                 }
 94:             }
 95:         } else {
 96:             $filter_list = array();
 97:         }
 98: 
 99:         // Factory setup
100:         $factory = new AssetFactory($asset_dir);
101: 
102:         $factory->setAssetManager($am);
103:         $factory->setFilterManager($fm);
104: 
105:         $factory->setDefaultOutput('*'.(! empty($asset_type) ? '.' : '').$asset_type);
106: 
107:         $factory->setDebug($debug);
108: 
109:         $factory->addWorker(new CacheBustingWorker());
110: 
111:         // Prepare the assets writer
112:         $writer = new AssetWriter($output_path);
113: 
114:         $asset = $factory->createAsset($asset_name, $filter_list);
115: 
116:         $cache = new AssetCache($asset, new FilesystemCache($output_path));
117: 
118:         $writer->writeAsset($cache);
119: 
120:         return rtrim($output_url, '/').'/'.$asset->getTargetPath();
121:     }
122: }
