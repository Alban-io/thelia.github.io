 68:     protected function initPropel()
 69:     {
 70:         if (file_exists(THELIA_ROOT . '/local/config/database.yml') === false) {
 71:             return ;
 72:         }
 73: 
 74:         $definePropel = new DefinePropel(new DatabaseConfiguration(),
 75:             Yaml::parse(THELIA_ROOT . '/local/config/database.yml'));
 76:         $propelConfig = $definePropel->getConfig();
 77:         $serviceContainer = Propel::getServiceContainer();
 78:         $serviceContainer->setAdapterClass('thelia', 'mysql');
 79:         $manager = new ConnectionManagerSingle();
 80:         $manager->setConfiguration($definePropel->getConfig());
 81:         $serviceContainer->setConnectionManager('thelia', $manager);
 82: 
 83:         if ($this->isDebug()) {
 84:             $serviceContainer->setLogger('defaultLogger', Tlog::getInstance());
 85: 
 86:             $con = Propel::getConnection(\Thelia\Model\Map\ProductTableMap::DATABASE_NAME);
 87:             $con->useDebug(true);
 88:         }
 89:     }

 93:      * Load some configuration
 94:      * Initialize all plugins
 95:      *
 96:      */
 97:     protected function loadConfiguration(ContainerBuilder $container)
 98:     {
 99: 
100:         $loader = new XmlFileLoader($container, new FileLocator(THELIA_ROOT . "/core/lib/Thelia/Config/Resources"));
101:         $loader->load("config.xml");
102:         $loader->load("routing.xml");
103:         $loader->load("action.xml");
104:         if (defined("THELIA_INSTALL_MODE") === false) {
105:             $modules = \Thelia\Model\ModuleQuery::getActivated();
106: 
107:             foreach ($modules as $module) {
108: 
109:                 try {
110:                     $loader = new XmlFileLoader($container, new FileLocator(THELIA_MODULE_DIR . "/" . ucfirst($module->getCode()) . "/Config"));
111:                     $loader->load("config.xml");
112:                 } catch (\InvalidArgumentException $e) {
113:                     // FIXME: process module configuration exception
114:                 }
115:             }
116:         }
117:     }
