 28: use Thelia\Tpex\Element\Loop\BaseLoop;
 29: use Thelia\Model\CategoryQuery;

 31: /**
 32:  *
 33:  * Category loop, all params available :
 34:  *
 35:  * - id : can be an id (eq : 3) or a "string list" (eg: 3, 4, 5)
 36:  * - parent : categories having this parent id
 37:  * - current : current id is used if you are on a category page
 38:  * - not_empty : if value is 1, category and subcategories must have at least 1 product
 39:  * - visible : default 1, if you want category not visible put 0
 40:  * - order : all value available :
 41:  *      * alpha : sorting by title alphabetical order
 42:  *      * alpha_reverse : sorting by title alphabetical reverse order
 43:  *      * reverse : sorting by position descending
 44:  *      * by default results are sorting by position ascending
 45:  * - random : if 1, random results. Default value is 0
 46:  * - exclude : all category id you want to exclude (as for id, an integer or a "string list" can be used)
 47:  * - limit : number of results. Default value is 10
 48:  * - offset : at witch id start the search
 49:  *
 50:  * example :
 51:  *
 52:  * <THELIA_cat type="category" parent="3" limit="4">
 53:  *      #TITLE : #ID
 54:  * </THELIA_cat>
 55:  *
 56:  *
 57:  * Class Category
 58:  * @package Thelia\Core\Template\Loop
 59:  * @author Manuel Raynaud <mraynaud@openstudio.fr>
 60:  */
 61: class Category extends BaseLoop {
 62: 
 63:     public $id;
 64:     public $parent;
 65:     public $current;
 66:     public $not_empty;
 67:     public $visible;
 68:     public $link;
 69:     public $order;
 70:     public $random;
 71:     public $exclude;
 72:     public $limit;
 73:     public $offset;
 74: 
 75:     public function defineArgs()
 76:     {
 77:         return array(
 78:             "id" => "optional",
 79:             "parent" => "optional",
 80:             "current" => "optional",
 81:             "not_empty" => 0,
 82:             "visible" => 1,
 83:             "link" => "optional",
 84:             "order" => "optional",
 85:             "random" => 0,
 86:             "exclude" => "optional",
 87:             "limit" => 10,
 88:             "offset" => 0,
 89:         );
 90:     }
 91: 
 92:     /**
 93:      *
 94:      *
 95:      * @param string $text
 96:      * @return mixed|string
 97:      */
 98:     public function exec($text)
 99:     {
100:         $search = CategoryQuery::create();
101: 
102:         if (!is_null($this->id)) {
103:             $search->filterById($this->id, \Criteria::IN);
104:         }
105: 
106:         if(!is_null($this->parent)) {
107:             $search->filterByParent($this->parent);
108:         }
109: 
110:         if($this->current == 1) {
111:             $search->filterById($this->request->get("category_id"));
112:         } else if ($this->current == 0) {
113:             $search->filterById($this->request->get("category_id"), \Criteria::NOT_IN);
114:         }
115: 
116:         if (!is_null($this->exclude)) {
117:             $search->filterById(explode(",", $this->exclude), \Criteria::NOT_IN);
118:         }
119: 
120:         if (!is_null($this->link)) {
121:             $search->filterByLink($this->link);
122:         }
123: 
124:         if($this->limit > -1) {
125:             $search->limit($this->limit);
126:         }
127:         $search->filterByVisible($this->visible);
128:         $search->offset($this->offset);
129: 
130: 
131:         switch($this->order) {
132:             case "alpha":
133:                 $search->addAscendingOrderByColumn(\Thelia\Model\CategoryI18nPeer::TITLE);
134:                 break;
135:             case "alpha_reverse":
136:                 $search->addDescendingOrderByColumn(\Thelia\Model\CategoryI18nPeer::TITLE);
137:                 break;
138:             case "reverse":
139:                 $search->orderByPosition(\Criteria::DESC);
140:                 break;
141:             default:
142:                 $search->orderByPosition();
143:                 break;
144:         }
145: 
146:         if($this->random == 1) {
147:             $search->clearOrderByColumns();
148:             $search->addAscendingOrderByColumn('RAND()');
149:         }
150: 
151:         /**
152:          * \Criteria::INNER_JOIN in second parameter for joinWithI18n  exclude query without translation.
153:          *
154:          * @todo : verify here if we want results for row without translations.
155:          */
156:         $search->joinWithI18n($this->request->getSession()->get('locale', 'en_US'), \Criteria::INNER_JOIN);
157: 
158:         $categories = $search->find();
159: 
160:         $res = "";
161: 
162:         foreach ($categories as $category) {
163: 
164:             if ($this->not_empty && $category->countAllProducts() == 0) continue;
165: 
166:             $temp = str_replace("#TITLE", $category->getTitle(), $text);
167:             $temp = str_replace("#CHAPO", $category->getChapo(), $temp);
168:             $temp = str_replace("#DESCRIPTION", $category->getDescription(), $temp);
169:             $temp = str_replace("#POSTSCRIPTUM", $category->getPostscriptum(), $temp);
170:             $temp = str_replace("#PARENT", $category->getParent(), $temp);
171:             $temp = str_replace("#ID", $category->getId(), $temp);
172:             $temp = str_replace("#URL", $category->getUrl(), $temp);
173:             $temp = str_replace("#LINK", $category->getLink(), $temp);
174:             $temp = str_replace("#NB_CHILD", $category->countChild(), $temp);
175:             $res .= $temp;
176:         }
177: 
178:         return $res;
179:     }
180: 
181: }
