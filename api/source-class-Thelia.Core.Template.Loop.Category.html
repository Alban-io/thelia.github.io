 40: /**
 41:  *
 42:  * Category loop, all params available :
 43:  *
 44:  * - id : can be an id (eq : 3) or a "string list" (eg: 3, 4, 5)
 45:  * - parent : categories having this parent id
 46:  * - current : current id is used if you are on a category page
 47:  * - not_empty : if value is 1, category and subcategories must have at least 1 product
 48:  * - visible : default 1, if you want category not visible put 0
 49:  * - order : all value available :  'alpha', 'alpha_reverse', 'manual' (default), 'manual-reverse', 'random'
 50:  * - exclude : all category id you want to exclude (as for id, an integer or a "string list" can be used)
 51:  *
 52:  * example :
 53:  *
 54:  * <THELIA_cat type="category" parent="3" limit="4">
 55:  *      #TITLE : #ID
 56:  * </THELIA_cat>
 57:  *
 58:  *
 59:  * Class Category
 60:  * @package Thelia\Core\Template\Loop
 61:  * @author Manuel Raynaud <mraynaud@openstudio.fr>
 62:  * @author Etienne Roudeix <eroudeix@openstudio.fr>
 63:  */
 64: class Category extends BaseLoop
 65: {
 66:     /**
 67:      * @return ArgumentCollection
 68:      */
 69:     protected function getArgDefinitions()
 70:     {
 71:         return new ArgumentCollection(
 72:             Argument::createIntListTypeArgument('id'),
 73:             Argument::createIntTypeArgument('parent'),
 74:             Argument::createBooleanTypeArgument('current'),
 75:             Argument::createBooleanTypeArgument('not_empty', 0),
 76:             Argument::createBooleanTypeArgument('visible', 1),
 77:             new Argument(
 78:                 'order',
 79:                 new TypeCollection(
 80:                     new Type\EnumListType(array('alpha', 'alpha_reverse', 'manual', 'manual-reverse', 'random'))
 81:                 ),
 82:                 'manual'
 83:             ),
 84:             Argument::createIntListTypeArgument('exclude')
 85:         );
 86:     }
 87: 
 88:     /**
 89:      * @param $pagination
 90:      *
 91:      * @return \Thelia\Core\Template\Element\LoopResult
 92:      */
 93:     public function exec(&$pagination)
 94:     {
 95:         $search = CategoryQuery::create();
 96: 
 97:         $id = $this->getId();
 98: 
 99:         if (!is_null($id)) {
100:             $search->filterById($id, Criteria::IN);
101:         }
102: 
103:         $parent = $this->getParent();
104: 
105:         if (!is_null($parent)) {
106:             $search->filterByParent($parent);
107:         }
108: 
109: 
110:         $current = $this->getCurrent();
111: 
112:         if ($current === true) {
113:             $search->filterById($this->request->get("category_id"));
114:         } elseif ($current === false) {
115:             $search->filterById($this->request->get("category_id"), Criteria::NOT_IN);
116:         }
117: 
118: 
119:          $exclude = $this->getExclude();
120: 
121:         if (!is_null($exclude)) {
122:             $search->filterById($exclude, Criteria::NOT_IN);
123:         }
124: 
125:         $search->filterByVisible($this->getVisible() ? 1 : 0);
126: 
127:         $orders  = $this->getOrder();
128: 
129:         foreach($orders as $order) {
130:             switch ($order) {
131:                 case "alpha":
132:                     $search->addAscendingOrderByColumn(\Thelia\Model\Map\CategoryI18nTableMap::TITLE);
133:                     break;
134:                 case "alpha_reverse":
135:                     $search->addDescendingOrderByColumn(\Thelia\Model\Map\CategoryI18nTableMap::TITLE);
136:                     break;
137:                 case "manual-reverse":
138:                     $search->orderByPosition(Criteria::DESC);
139:                     break;
140:                 case "manual":
141:                     $search->orderByPosition(Criteria::ASC);
142:                     break;
143:                 case "random":
144:                     $search->clearOrderByColumns();
145:                     $search->addAscendingOrderByColumn('RAND()');
146:                     break(2);
147:                     break;
148:             }
149:         }
150: 
151:         /**
152:          * \Criteria::INNER_JOIN in second parameter for joinWithI18n  exclude query without translation.
153:          *
154:          * @todo : verify here if we want results for row without translations.
155:          */
156: 
157:         $search->joinWithI18n(
158:             $this->request->getSession()->getLocale(),
159:             (ConfigQuery::read("default_lang_without_translation", 1)) ? Criteria::LEFT_JOIN : Criteria::INNER_JOIN
160:         );
161: 
162:         $categories = $this->search($search, $pagination);
163: 
164:         $loopResult = new LoopResult();
165: 
166:         foreach ($categories as $category) {
167: 
168:             if ($this->not_empty && $category->countAllProducts() == 0) continue;
169: 
170:             $loopResultRow = new LoopResultRow();
171:             $loopResultRow->set("TITLE",$category->getTitle());
172:             $loopResultRow->set("CHAPO", $category->getChapo());
173:             $loopResultRow->set("DESCRIPTION", $category->getDescription());
174:             $loopResultRow->set("POSTSCRIPTUM", $category->getPostscriptum());
175:             $loopResultRow->set("PARENT", $category->getParent());
176:             $loopResultRow->set("ID", $category->getId());
177:             $loopResultRow->set("URL", $category->getUrl());
178:             $loopResultRow->set("PRODUCT_COUNT", $category->countChild());
179: 
180:             $loopResult->addRow($loopResultRow);
181:         }
182: 
183:         return $loopResult;
184:     }
185: }
