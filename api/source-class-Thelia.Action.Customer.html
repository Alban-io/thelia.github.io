 57:     public function create(ActionEvent $event)
 58:     {
 59:         $request = $event->getRequest();
 60: 
 61:         $customerCreationForm = new CustomerCreation($request);
 62: 
 63:         $form = $customerCreationForm->getForm();
 64: 
 65:         if ($request->isMethod("post")) {
 66: 
 67:             $form->bind($request);
 68: 
 69:             if ($form->isValid()) {
 70:                 $data = $form->getData();
 71:                 $customer = new CustomerModel();
 72: 
 73:                 try {
 74:                     $event->getDispatcher()->dispatch(TheliaEvents::BEFORE_CREATECUSTOMER, $event);
 75: 
 76:                     $customer->createOrUpdate(
 77:                         $data["title"],
 78:                         $data["firstname"],
 79:                         $data["lastname"],
 80:                         $data["address1"],
 81:                         $data["address2"],
 82:                         $data["address3"],
 83:                         $data["phone"],
 84:                         $data["cellphone"],
 85:                         $data["zipcode"],
 86:                         $data["country"],
 87:                         $data["email"],
 88:                         $data["password"],
 89:                         $request->getSession()->getLang()
 90:                     );
 91: 
 92:                     $event->getDispatcher()->dispatch(TheliaEvents::AFTER_CREATECUSTOMER, $event);
 93: 
 94:                     // Connect the newly created user,and redirect to the success URL
 95:                     $this->processSuccessfulLogin($event, $customer, $customerCreationForm, true);
 96: 
 97:                 } catch (PropelException $e) {
 98:                     Tlog::getInstance()->error(sprintf('error during creating customer on action/createCustomer with message "%s"', $e->getMessage()));
 99: 
100:                     $message = "Failed to create your account, please try again.";
101:                 }
102:             }
103:             else {
104:                 $message = "Missing or invalid data";
105:             }
106:         }
107:         else {
108:             $message = "Wrong form method !";
109:         }
110: 
111:         // The form has an error
112:         $customerCreationForm->setError(true);
113:         $customerCreationForm->setErrorMessage($message);
114: 
115:         // Store the form in the parser context
116:         $event->setErrorForm($customerCreationForm);
117: 
118:         // Stop event propagation
119:         $event->stopPropagation();
120:     }

198:     /**
199:      * Perform user login. On a successful login, the user is redirected to the URL
200:      * found in the success_url form parameter, or / if none was found.
201:      *
202:      * If login is not successfull, the same view is dispolyed again.
203:      *
204:      * @param ActionEvent $event
205:      */
206:     public function login(ActionEvent $event)
207:     {
208:         $request = $event->getRequest();
209: 
210:         $customerLoginForm = new CustomerLogin($request);
211: 
212:         $authenticator = new CustomerUsernamePasswordFormAuthenticator($request, $customerLoginForm);
213: 
214:         try {
215:             $user = $authenticator->getAuthentifiedUser();
216: 
217:             $this->processSuccessfulLogin($event, $user, $customerLoginForm);
218:         }
219:         catch (ValidatorException $ex) {
220:             $message = "Missing or invalid information. Please check your input.";
221:         }
222:         catch (UsernameNotFoundException $ex) {
223:             $message = "This email address was not found.";
224:         }
225:         catch (AuthenticationException $ex) {
226:             $message = "Login failed. Please check your username and password.";
227:         }
228:         catch (\Exception $ex) {
229:             $message = sprintf("Unable to process your request. Please try again (%s in %s).", $ex->getMessage(), $ex->getFile());
230:         }
231: 
232:         // The for has an error
233:         $customerLoginForm->setError(true);
234:         $customerLoginForm->setErrorMessage($message);
235: 
236:         // Dispatch the errored form
237:         $event->setErrorForm($customerLoginForm);
238: 
239:         // A this point, the same view is displayed again.
240:     }

283:     protected function processSuccessfulLogin(ActionEvent $event, CustomerModel $user, BaseForm $form, $sendLoginEvent = false)
284:     {

288:         if ($sendLoginEvent) $event->getDispatcher()->dispatch(TheliaEvents::CUSTOMER_LOGIN, $event);
